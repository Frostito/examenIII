import React, { useState } from 'react';

interface Panel {
  name: string;
  status: string;
}

const PanelForm = () => {
  const [newPanelName, setNewPanelName] = useState('');
  const [panels, setPanels] = useState<Panel[]>([]);

  const handleAddPanel = () => {
    const trimmedPanelName = newPanelName.trim();
    if (trimmedPanelName === '') {
      return;
    }
    if (panels.some((panel) => panel.name.trim().toLowerCase() === trimmedPanelName.toLowerCase())) {
      return;
    }
    const newPanel = { name: trimmedPanelName, status: 'Nuevo' };
    setPanels([...panels, newPanel]);
    setNewPanelName('');
  };

  const handlePanelNameChange = (event: React.ChangeEvent<HTMLInputElement>) => {
    setNewPanelName(event.target.value);
  };

  return (
    <div>
      <h1>Mis paneles</h1>
      <div>
        <input type="text" value={newPanelName} onChange={handlePanelNameChange} />
        <button onClick={handleAddPanel} disabled={!newPanelName.trim()}>
          Agregar
        </button>
      </div>
      <ul>
        {panels.map((panel) => (
          <li key={panel.name}>
            {panel.name} - {panel.status}
          </li>
        ))}
      </ul>
    </div>
  );
};

export default PanelForm;
